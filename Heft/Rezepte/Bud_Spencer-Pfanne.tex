\prorezeptzutaten{
	\item 200 Gramm gewürfelter Speck
	\item 1 Zwiebel
	\item 1 Dose gehackte Tomaten
	\item 1 Dose weiße Bohnen
	\item 1 Dose Kidneybohnen
	\item 160 Gramm saure Sahne oder Joghurt
	\item Salz und Pfeffer
	\item Thymian, Chiligewürz, Cayenne-Pfeffer
}
\prorezeptequipment{
	\item Pfanne
	\item Herdplatte (alternativ: Gaskocher im Keller)

	\propostqrinline{427695}
}

\prorezeptzubereitung{
	\item Zwiebeln pellen und rein hacken.
	Zusammen mit dem Speck und den in Scheiben geschnittenen Cabanossi anbraten. Ruhig kross anbraten.
	\item Bohnen abtropfen lassen und in die Pfanne geben, kurz, mitbraten.
	\item Tomaten dazugeben und gut durchrühren.
	\item Mit Salz, Pfeffer und den anderen Gewürzen abschmecken.
	\item Zum Schluss die saure Sahne/Joghurt unterrühren.
}

	Bacon Bomb
}{
	TankOne
}{
	459660
}{
	\diffaltschwuchtel
}

\hspace{10mm}

\prorezeptzutaten{
	\item 500g Hackfleisch (gemischt, Rind, Schwein, Elefant, Kadse oder müffelnde Oma, jeder wie er mag)
	\item 3-4 Packungen Bacon
	\item 1 Packung BBQ-Soße
	\item 1 Zwiebel
	\item 1 Ei
	\item Paniermehl
	\item Salz/Pfeffer
	\item Schnittlauch
	\item Optional: Knoblauch, Tomatenmark, Olivenöl, Käse, etc.
}
\prorezeptequipment{
	\item Schüssel
	\item Backofen oder Grill
	\item Optional: Pinsel

	\propostqrinline{459660}
}

\prorezeptzubereitung{
	\item Bacon-Teppich weben (davon gibt's 'ne Million Anleitungen im Internet, ich gehe davon aus, dass die Hälfte der Kellerkinder das sowieso schon kennt oder schnell googlen kann, daher geb' ich dafür keine konkrete Anleitung, sondern bloß 'nen paar Bilder, weil Bacon-pr0n und so).
	\item Hackfleisch, Paniermehl und das Ei (natürlich nur den Inhalt) vermengen.
	\item Die optionalen Zutaten ebenfalls mit vermengen.
	\item Zwiebel, Schnittlauch und den Rest vom Bacon und die BBQ-Soße drauf (hier könt Ihr noch dazumischen, was Ihr mögt).
	\item Den Kram als Klumpen auf das Bacon-Gewebe packen und im Bacon-Teppich einwickeln.
	\item Bei \texttt{150\celsius} Umluft etwas über eine Stunde backen. Grill ist geiler, aber dafür müsste man halt rausgehen. Zwischendrin noch mal mit BBQ-Soße bepinseln.
}
